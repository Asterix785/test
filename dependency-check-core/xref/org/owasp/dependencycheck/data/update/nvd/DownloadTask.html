25  import java.util.concurrent.ExecutorService;

27  import javax.annotation.concurrent.ThreadSafe;
28  import org.apache.commons.lang3.StringUtils;
29  import org.owasp.dependencycheck.data.nvdcve.CveDB;
30  import org.owasp.dependencycheck.data.update.exception.UpdateException;
31  import org.owasp.dependencycheck.utils.DownloadFailedException;
32  import org.owasp.dependencycheck.utils.Downloader;
33  import org.owasp.dependencycheck.utils.ResourceNotFoundException;
34  import org.owasp.dependencycheck.utils.Settings;

36  import org.slf4j.Logger;
37  import org.slf4j.LoggerFactory;

39  /**
40   * A callable object to download two files.
41   *
42   * @author Jeremy Long
43   */
44  @ThreadSafe
45  public class DownloadTask implements Callable<Future<ProcessTask>> {
46  
47      /**
48       * The Logger.
49       */
50      private static final Logger LOGGER = LoggerFactory.getLogger(DownloadTask.class);
51      /**
52       * The CVE DB to use when processing the files.
53       */
54      private final CveDB cveDB;
55      /**
56       * The processor service to pass the results of the download to.
57       */
58      private final ExecutorService processorService;
59      /**
60       * The NVD CVE Meta Data.
61       */
62      private NvdCveInfo nvdCveInfo;
63      /**
64       * A reference to the global settings object.
65       */
66      private final Settings settings;
67      /**
68       * a file.
69       */
70      private final File file;
71  
72      /**
73       * Simple constructor for the callable download task.
74       *
75       * @param nvdCveInfo the NVD CVE info
76       * @param processor the processor service to submit the downloaded files to
77       * @param cveDB the CVE DB to use to store the vulnerability data
78       * @param settings a reference to the global settings object; this is
79       * necessary so that when the thread is started the dependencies have a
80       * correct reference to the global settings.
81       * @throws UpdateException thrown if temporary files could not be created
82       */
83      public DownloadTask(NvdCveInfo nvdCveInfo, ExecutorService processor, CveDB cveDB, Settings settings) throws UpdateException {
84          this.nvdCveInfo = nvdCveInfo;
85          this.processorService = processor;
86          this.cveDB = cveDB;
87          this.settings = settings;
88  
89          try {
90              this.file = File.createTempFile("cve" + nvdCveInfo.getId() + '_', ".json.gz", settings.getTempDirectory());
91          } catch (IOException ex) {
92              throw new UpdateException("Unable to create temporary files", ex);
93          }
94      }
95  
96      /**
97       * Get the value of nvdCveInfo.
98       *
99       * @return the value of nvdCveInfo
100      */
101     public NvdCveInfo getNvdCveInfo() {
102         return nvdCveInfo;
103     }
104 
105     /**
106      * Set the value of nvdCveInfo.
107      *
108      * @param nvdCveInfo new value of nvdCveInfo
109      */
110     public void setNvdCveInfo(NvdCveInfo nvdCveInfo) {
111         this.nvdCveInfo = nvdCveInfo;
112     }
113 
114     /**
115      * Get the value of file.
116      *
117      * @return the value of file
118      */
119     public File getFile() {
120         return file;
121     }
122 
123     @SuppressWarnings("BusyWait")
124     @Override
125     public Future<ProcessTask> call() throws Exception {
126         final long waitTime = settings.getInt(Settings.KEYS.CVE_DOWNLOAD_WAIT_TIME, 4000);
127         long startDownload = 0;
128         final NvdCache cache = new NvdCache(settings);
129         try {
130             final URL url1 = new URL(nvdCveInfo.getUrl());
131             if (cache.notInCache(url1, file)) {
132                 Thread.sleep(waitTime);
133                 LOGGER.info("Download Started for NVD CVE - {}", nvdCveInfo.getId());
134                 startDownload = System.currentTimeMillis();
135                 final int downloadAttempts = 5;
136                 for (int x = 2; x <= downloadAttempts && !attemptDownload(url1, x == downloadAttempts); x++) {
137                     LOGGER.info("Download Attempt {} for NVD CVE - {}", x, nvdCveInfo.getId());
138                     Thread.sleep(waitTime * (x / 2));
139                 }
140                 if (file.isFile() && file.length() > 0) {
141                     LOGGER.info("Download Complete for NVD CVE - {}  ({} ms)", nvdCveInfo.getId(),
142                             System.currentTimeMillis() - startDownload);
143                     cache.storeInCache(url1, file);
144                 } else {
145                     throw new DownloadFailedException("Unable to download NVD CVE " + nvdCveInfo.getId());
146                 }
147             }
148             if (this.processorService == null) {
149                 return null;
150             }
151             final ProcessTask task = new ProcessTask(cveDB, this, settings);
152             final Future<ProcessTask> val = this.processorService.submit(task);
153             return val;
154 
155         } catch (Throwable ex) {
156             LOGGER.error("Error downloading NVD CVE - {} Reason: {}", nvdCveInfo.getId(), ex.getMessage());
157             throw ex;
158         } finally {
159             settings.cleanup(false);
160         }
161     }
162 
163     private boolean attemptDownload(final URL url1, boolean showLog) throws TooManyRequestsException, ResourceNotFoundException {
164         try {
165             final Downloader downloader = new Downloader(settings);
166             downloader.fetchFile(url1, file, Settings.KEYS.CVE_USER, Settings.KEYS.CVE_PASSWORD);
167         } catch (DownloadFailedException ex) {
168             if (showLog) {
169                 LOGGER.error("Download Failed for NVD CVE - {}\nSome CVEs may not be reported. Reason: {}",
170                         nvdCveInfo.getId(), ex.getMessage());
171                 if (settings.getString(Settings.KEYS.PROXY_SERVER) == null) {
172                     LOGGER.error("If you are behind a proxy you may need to configure dependency-check to use the proxy.");
173                 }
174                 LOGGER.debug("", ex);
175             }
176             return false;
177         }
178         return true;
179     }
180 
181     /**
182      * Attempts to delete the files that were downloaded.
183      */
184     public void cleanup() {
185         if (file != null && file.exists() && !file.delete()) {
186             LOGGER.debug("Failed to delete first temporary file {}", file);
187             file.deleteOnExit();
188         }
189     }
190 
191     /**
192      * Attempts to delete the files that were downloaded.
193      */
194     public void evictCorruptFileFromCache() {
195         final NvdCache cache = new NvdCache(settings);
196         try {
197             final URL url1 = new URL(nvdCveInfo.getUrl());
198             cache.evictFromCache(url1);
199         } catch (MalformedURLException e) {
200             LOGGER.debug("Ignoring Cache-eviction request for an invalid URL");
201         }
202     }
203 
204     /**
205      * Returns true if the process task is for the modified json file from the
206      * NVD.
207      *
208      * @return <code>true</code> if the process task is for the modified data;
209      * otherwise <code>false</code>
210      */
211     public boolean isModified() {
212         return StringUtils.containsIgnoreCase(file.toString(), "modified");
213     }
